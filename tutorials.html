Example 1. Bubble Sort

          
  def bubble_sort(array)
      n = array.length
      loop do
        swapped = false
        # Iterate from 0 to n-2
        (1...n).each do |i|
          # Compare the current element with the previous one
          if array[i - 1] > array[i]
            # Swap the elements
            array[i - 1], array[i] = array[i], array[i - 1]
            swapped = true
          end
        end
        # Break the loop if no elements were swapped
        break unless swapped
      end
      # Return sorted array
      array
    end
    # Example usage:
    arr = [5, 3, 8, 4, 2]
    puts "Original array: #{arr}"
    sorted_arr = bubble_sort(arr)
    puts "Sorted array: #{sorted_arr}"

Example 2. Classes

          
  # Base class representing a general greeter
  class Greeter
      # Constructor method: Called when a new instance of the class is created
      # Initializes the greeter with a name
      def initialize(name)
        @name = name  # Instance variable to store the name
      end
    
      # Method to greet using the name
      def greet
        puts "Hello, #{@name}!"  # Output a general greeting message
      end
    end
    
    # Subclass of Greeter specifically for Walmart greeters
    class WalmartGreeter < Greeter
      # Additional method for Walmart-specific greeting
      def store_greet
        # Output a store-specific greeting message
        puts "Hello, #{@name}, welcome to Walmart!"
      end
    end
    
    # Creating an instance of the Greeter class
    greeter = Greeter.new("Ruby")
    greeter.greet  
    
    # Creating an instance of the WalmartGreeter subclass
    walmart_greeter = WalmartGreeter.new("Matz")  
    walmart_greeter.greet 
    walmart_greeter.store_greet
